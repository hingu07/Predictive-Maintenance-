#include <Wire.h>
#include <MPU6050.h>
#include <DHT.h>

// ── Sensor Objects ────────────────────────────────────────
MPU6050 mpu;
#define DHTPIN    3
#define DHTTYPE   DHT11
DHT dht(DHTPIN, DHTTYPE);

// ── Pin Definitions ───────────────────────────────────────
#define CURRENT_PIN  A1
#define SOUND_PIN    A0
#define BUZZ_CURRENT 6
#define BUZZ_SOUND   5
#define BUZZ_VIBE    7
#define BUZZ_TEMP    4

// ── Default Thresholds (can be updated via Serial commands) ─
float CURRENT_THRESHOLD   = 1.00;   // Amps
int   SOUND_THRESHOLD     = 15;     // Peak-to-peak ADC counts
float VIBRATION_THRESHOLD = 0.08;   // g
float TEMP_THRESHOLD      = 33.0;   // °C

// ── ACS712 Constants ──────────────────────────────────────
const float SENSITIVITY   = 0.185;  // V/A for ACS712-5A
// Dynamic zero-offset calibration (measured at startup)
int ZERO_OFFSET = 512;

// ── MPU6050 Baseline ─────────────────────────────────────
long  baseAx = 0, baseAy = 0, baseAz = 0;
bool  calibrated = false;

// ── Vibration Rolling Average ─────────────────────────────
#define VIBE_HISTORY  10
float vibeHistory[VIBE_HISTORY];
int   vibeIdx        = 0;
float vibeRollingAvg = 0.0;

// ── DHT ───────────────────────────────────────────────────
float lastTemp  = 0.0;
float lastHumid = 0.0;
unsigned long lastDHTRead = 0;

// ── Non-blocking Timers ───────────────────────────────────
unsigned long tCurrentStart  = 0;
unsigned long tSoundStart    = 0;
unsigned long tVibeStart     = 0;

// Current RMS accumulators
long  currSumSq = 0;
int   currN     = 0;
float currentA  = 0.0;
bool  currAlert = false;

// Sound peak-to-peak accumulators
int   sndMax = 0, sndMin = 1023;
int   volumeLevel = 0;
bool  soundAlert  = false;

// Vibration RMS accumulators
long long vibeSumSq = 0;
int       vibeN     = 0;
bool      vibeAlert = false;

// ── Output Timer ─────────────────────────────────────────
unsigned long lastOutput = 0;

// ── Serial Command Buffer ─────────────────────────────────
String cmdBuffer = "";

// ============================================================
// Calibrate ACS712 zero-offset (machine idle, no load)
// ============================================================
void calibrateCurrentSensor() {
  long sum = 0;
  for (int i = 0; i < 100; i++) {
    sum += analogRead(CURRENT_PIN);
    delayMicroseconds(200);
  }
  ZERO_OFFSET = sum / 100;
}

// ============================================================
// Calibrate MPU6050 static gravity baseline
// Uses minimal delay — just microsecond waits between reads
// ============================================================
void calibrateMPU() {
  long sumX = 0, sumY = 0, sumZ = 0;
  const int SAMPLES = 200;
  for (int i = 0; i < SAMPLES; i++) {
    int16_t ax, ay, az;
    mpu.getAcceleration(&ax, &ay, &az);
    sumX += ax; sumY += ay; sumZ += az;
    delayMicroseconds(500);  // ~0.5ms each = ~100ms total (was 2000ms!)
  }
  baseAx = sumX / SAMPLES;
  baseAy = sumY / SAMPLES;
  baseAz = sumZ / SAMPLES;
  calibrated = true;
}

// ============================================================
// Parse threshold-update commands from Python
// Format:  SET_TEMP:33.5   SET_AMP:1.5   SET_VIBE:0.10   SET_SND:20
// ============================================================
void parseCommand(String cmd) {
  cmd.trim();
  if (cmd.startsWith("SET_TEMP:")) {
    TEMP_THRESHOLD = cmd.substring(9).toFloat();
  } else if (cmd.startsWith("SET_AMP:")) {
    CURRENT_THRESHOLD = cmd.substring(8).toFloat();
  } else if (cmd.startsWith("SET_VIBE:")) {
    VIBRATION_THRESHOLD = cmd.substring(9).toFloat();
  } else if (cmd.startsWith("SET_SND:")) {
    SOUND_THRESHOLD = cmd.substring(8).toInt();
  } else if (cmd == "RECAL") {
    calibrateMPU();
    calibrateCurrentSensor();
    Serial.println("RECAL_DONE");
  }
}

// ============================================================
void setup() {
  Serial.begin(115200);   // ← 115200 baud — much faster, less lag
  Wire.begin();
  Wire.setClock(400000);  // 400 kHz I²C (was default 100 kHz) — faster MPU reads

  mpu.initialize();
  if (!mpu.testConnection()) {
    Serial.println("ERR:MPU6050");
    while (1);
  }
  mpu.setFullScaleAccelRange(MPU6050_ACCEL_FS_2);  // ±2g → 16384 LSB/g
  mpu.setDLPFMode(MPU6050_DLPF_BW_98);

  dht.begin();

  pinMode(BUZZ_CURRENT, OUTPUT);
  pinMode(BUZZ_SOUND,   OUTPUT);
  pinMode(BUZZ_VIBE,    OUTPUT);
  pinMode(BUZZ_TEMP,    OUTPUT);

  for (int i = 0; i < VIBE_HISTORY; i++) vibeHistory[i] = 0;

  // Force a DHT read immediately so we don't wait 2s for first data
  lastDHTRead = millis() - 2000;

  calibrateCurrentSensor();
  calibrateMPU();

  // Tell Python we're ready + current thresholds
  Serial.println("READY");
  Serial.print("THRESH:");
  Serial.print(TEMP_THRESHOLD, 1); Serial.print(",");
  Serial.print(CURRENT_THRESHOLD, 2); Serial.print(",");
  Serial.print(VIBRATION_THRESHOLD, 4); Serial.print(",");
  Serial.println(SOUND_THRESHOLD);
  Serial.println("FORMAT:TEMP,HUMID,VIBE,AMP,SOUND,VA,CA,SA,TA");
}

// ============================================================
void loop() {

  // ── Read incoming Serial commands (non-blocking) ──────
  while (Serial.available()) {
    char c = Serial.read();
    if (c == '\n') {
      if (cmdBuffer.length() > 0) {
        parseCommand(cmdBuffer);
        cmdBuffer = "";
      }
    } else {
      cmdBuffer += c;
    }
  }

  unsigned long now = millis();

  // ══ 1. CURRENT — sample continuously, compute every 50ms ══
  {
    int raw = analogRead(CURRENT_PIN);
    long c  = (long)raw - ZERO_OFFSET;
    currSumSq += c * c;
    currN++;
    if (now - tCurrentStart >= 50) {
      if (currN > 0) {
        float rmsRaw = sqrt((float)currSumSq / currN);
        currentA = (rmsRaw * 5.0 / 1024.0) / SENSITIVITY;
        if (currentA < 0.03) currentA = 0.00;  // dead-band noise floor
        currAlert = (currentA >= CURRENT_THRESHOLD);
        digitalWrite(BUZZ_CURRENT, currAlert ? HIGH : LOW);
      }
      currSumSq     = 0;
      currN         = 0;
      tCurrentStart = now;
    }
  }

  // ══ 2. SOUND — sample continuously, compute every 50ms ══
  {
    int v = analogRead(SOUND_PIN);
    if (v > sndMax) sndMax = v;
    if (v < sndMin) sndMin = v;
    if (now - tSoundStart >= 50) {
      volumeLevel = sndMax - sndMin;
      soundAlert  = (volumeLevel >= SOUND_THRESHOLD);
      digitalWrite(BUZZ_SOUND, soundAlert ? HIGH : LOW);
      sndMax = 0; sndMin = 1023;
      tSoundStart = now;
    }
  }

  // ══ 3. VIBRATION — sample continuously, compute every 100ms ══
  {
    if (calibrated) {
      int16_t ax, ay, az;
      mpu.getAcceleration(&ax, &ay, &az);
      long dx = (long)ax - baseAx;
      long dy = (long)ay - baseAy;
      long dz = (long)az - baseAz;
      vibeSumSq += ((long long)dx*dx + (long long)dy*dy + (long long)dz*dz) / 16384LL;
      vibeN++;
    }
    if (now - tVibeStart >= 100) {
      if (vibeN > 0) {
        float vibeRMS = sqrt((float)vibeSumSq / vibeN) / 128.0;
        // Rolling average
        vibeRollingAvg -= vibeHistory[vibeIdx] / (float)VIBE_HISTORY;
        vibeHistory[vibeIdx] = vibeRMS;
        vibeRollingAvg += vibeRMS / (float)VIBE_HISTORY;
        vibeIdx = (vibeIdx + 1) % VIBE_HISTORY;
        vibeAlert = (vibeRollingAvg >= VIBRATION_THRESHOLD);
        digitalWrite(BUZZ_VIBE, vibeAlert ? HIGH : LOW);
      }
      vibeSumSq = 0;
      vibeN     = 0;
      tVibeStart = now;
    }
  }

  // ══ 4. DHT11 — read every 2s (sensor limit) ══
  if (now - lastDHTRead >= 2000) {
    float t = dht.readTemperature();
    float h = dht.readHumidity();
    lastDHTRead = now;
    if (!isnan(t) && !isnan(h)) {
      lastTemp  = t;
      lastHumid = h;
    }
    bool tempAlert = (lastTemp >= TEMP_THRESHOLD);
    digitalWrite(BUZZ_TEMP, tempAlert ? HIGH : LOW);
  }

  // ══ 5. Output CSV every 500ms ══
  // Sending at 2Hz gives Python smooth updates without flooding serial
  if (now - lastOutput >= 500) {
    lastOutput = now;
    bool tempAlert = (lastTemp >= TEMP_THRESHOLD);

    Serial.print(lastTemp, 1);        Serial.print(",");
    Serial.print(lastHumid, 1);       Serial.print(",");
    Serial.print(vibeRollingAvg, 4);  Serial.print(",");
    Serial.print(currentA, 3);        Serial.print(",");
    Serial.print(volumeLevel);        Serial.print(",");
    Serial.print(vibeAlert  ? 1 : 0); Serial.print(",");
    Serial.print(currAlert  ? 1 : 0); Serial.print(",");
    Serial.print(soundAlert ? 1 : 0); Serial.print(",");
    Serial.println(tempAlert  ? 1 : 0);
  }
}

class EventLog:
    def __init__(self, maxlen=80):
        self.lock    = threading.Lock()
        self._events = deque(maxlen=maxlen)

    def add(self, kind, key, value, severity):
        with self.lock:
            self._events.append({
                "time": time.time(), "kind": kind,
                "key": key, "value": value, "severity": severity,
            })

    def get_all(self):
        with self.lock:
            return list(self._events)

class DataStore:
    def __init__(self, maxlen=HISTORY_LEN):
        self.lock   = threading.Lock()
        self.times  = deque(maxlen=maxlen)
        self.temp   = deque(maxlen=maxlen)
        self.humid  = deque(maxlen=maxlen)
        self.vibe   = deque(maxlen=maxlen)
        self.amp    = deque(maxlen=maxlen)
        self.sound  = deque(maxlen=maxlen)
        self.health = deque(maxlen=maxlen)
        self.alerts = {k: 0.0 for k in ["temp","vibe","amp","sound"]}
        self._prev  = {k: False for k in ["temp","vibe","amp","sound"]}
        self.connected = False
        self.last_row  = None
        self.event_log = EventLog()

    def push(self, row):
        t, h, v, a, s, va, ca, sa, ta = row
        now   = time.time()
        score = self._health(t, v, a, s)
        amap  = {"vibe": bool(va), "amp": bool(ca), "sound": bool(sa), "temp": bool(ta)}
        vals  = {"temp": t, "vibe": v, "amp": a, "sound": s}
        with self.lock:
            self.times.append(now); self.temp.append(t); self.humid.append(h)
            self.vibe.append(v);    self.amp.append(a);  self.sound.append(s)
            self.health.append(score)
            for k, active in amap.items():
                if active:
                    self.alerts[k] = now
                    if not self._prev[k]:
                        ratio = vals[k]/THRESHOLDS[k] if THRESHOLDS[k] else 0
                        sev = "CRITICAL" if ratio > 1.5 else "HIGH" if ratio > 1.2 else "MEDIUM"
                        self.event_log.add("alert", k, vals[k], sev)
                elif self._prev[k]:
                    self.event_log.add("recovery", k, vals[k], "OK")
                self._prev[k] = active
            self.last_row = row

    def _health(self, t, v, a, s):
        def r(val, thr): return min(val/thr, 1.0) if thr > 0 else 0
        pen = (
            HEALTH_WEIGHTS["temp"]  * r(max(t-20, 0), max(THRESHOLDS["temp"]-20, 1)) +
            HEALTH_WEIGHTS["vibe"]  * r(v, THRESHOLDS["vibe"]) +
            HEALTH_WEIGHTS["amp"]   * r(a, THRESHOLDS["amp"]) +
            HEALTH_WEIGHTS["sound"] * r(s, THRESHOLDS["sound"])
        )
        return max(0, 100 - pen*100)

    def get_snapshot(self):
        with self.lock:
            return {
                "times": list(self.times), "temp": list(self.temp),
                "humid": list(self.humid), "vibe": list(self.vibe),
                "amp":   list(self.amp),   "sound": list(self.sound),
                "health":list(self.health),"alerts": dict(self.alerts),
                "last":  self.last_row,    "events": self.event_log.get_all(),
            }

class SerialReader(threading.Thread):
    def __init__(self, port, store, log_writer, on_error):
        super().__init__(daemon=True)
        self.port=port; self.store=store; self.log_writer=log_writer
        self.on_error=on_error; self.running=True; self._ser=None
        self._send_lock = threading.Lock()

    def send_command(self, cmd):
        try:
            if self._ser and self._ser.is_open:
                with self._send_lock:
                    self._ser.write((cmd.strip()+"\n").encode("utf-8"))
        except Exception: pass

    def run(self):
        try:
            self._ser = serial.Serial(self.port, BAUD_RATE, timeout=2)
            time.sleep(2); self._ser.reset_input_buffer()
            self.store.connected = True
            while self.running:
                try:
                    raw = self._ser.readline()
                    if not raw: continue
                    line = raw.decode("utf-8", errors="ignore").strip()
                except serial.SerialException: break
                if not line or any(line.startswith(p) for p in
                        ["FORMAT","MACHINE","READY","THRESH","RECAL","ERR","Calib"]):
                    continue
                parts = line.split(",")
                if len(parts) != 9: continue
                try:
                    row = (float(parts[0]), float(parts[1]), float(parts[2]),
                           float(parts[3]), int(float(parts[4])),
                           int(parts[5]), int(parts[6]), int(parts[7]), int(parts[8]))
                    if not (-10 < row[0] < 120): continue
                    if not (0 <= row[1] <= 100): continue
                    if row[3] < 0: continue
                    self.store.push(row); self.log_writer(row)
                except (ValueError, IndexError): continue
        except serial.SerialException as e:
            self.on_error(str(e))
        finally:
            self.store.connected = False
            if self._ser and self._ser.is_open:
                try: self._ser.close()
                except Exception: pass

    def stop(self): self.running = False

class InteractiveChart(tk.Canvas):
    """Chart with hover crosshair, tooltip, threshold line, overlay series."""
    def __init__(self, parent, color, threshold=None, label="", unit="", **kw):
        super().__init__(parent, bg=T["PANEL"], highlightthickness=0, **kw)
        self.color=color; self.threshold=threshold
        self.label=label; self.unit=unit
        self.data=[]; self.times_data=[]; self.overlay=[]
        self._hover_x = None
        self.bind("<Configure>", lambda e: self.redraw())
        self.bind("<Motion>",    self._on_move)
        self.bind("<Leave>",     self._on_leave)

    def update_data(self, data, times=None, threshold=None, overlay=None):
        self.data=list(data); self.times_data=list(times) if times else []
        if threshold is not None: self.threshold=threshold
        if overlay   is not None: self.overlay=overlay
        self.redraw()

    def _on_move(self, e): self._hover_x=e.x; self.redraw()
    def _on_leave(self, e): self._hover_x=None; self.redraw()

    def redraw(self):
        self.config(bg=T["PANEL"]); self.delete("all")
        w,h=self.winfo_width(),self.winfo_height()
        if w<20 or h<20: return
        PL,PR,PT,PB = 40,12,10,22
        iw=w-PL-PR; ih=h-PT-PB

        all_vals = list(self.data)
        for ov in self.overlay: all_vals += list(ov.get("data",[]))
        if self.threshold is not None: all_vals.append(self.threshold)
        if not all_vals:
            self.create_text(w//2,h//2,text="— waiting —",fill=T["SUBTEXT"],font=("Consolas",9))
            return

        dmin=min(all_vals); dmax=max(all_vals)
        span=(dmax-dmin) if dmax!=dmin else max(abs(dmax)*0.2,0.001)
        dmin-=span*0.08; dmax+=span*0.08; span=dmax-dmin

        def xt(i,n): return PL+iw*i/max(n-1,1)
        def yt(v):   return PT+ih*(1-(v-dmin)/span)

        # Grid + Y labels
        for i in range(5):
            frac=i/4; y=PT+ih*frac; val=dmax-span*frac
            self.create_line(PL,y,w-PR,y,fill=T["GRID"],width=1,dash=(2,4))
            self.create_text(PL-4,y,text=f"{val:.2g}",anchor="e",
                             fill=T["SUBTEXT"],font=("Consolas",7))

        n=len(self.data)
        if n>=2:
            for frac in [0,0.25,0.5,0.75,1.0]:
                xi=int(frac*(n-1)); x=xt(xi,n)
                if self.times_data and xi<len(self.times_data):
                    ago=int(time.time()-self.times_data[xi])
                    lbl=f"-{ago}s" if ago>0 else "now"
                else: lbl=""
                self.create_text(x,h-5,text=lbl,fill=T["SUBTEXT"],font=("Consolas",7))

        self.create_line(PL,PT,PL,h-PB,fill=T["BORDER"],width=1)
        self.create_line(PL,h-PB,w-PR,h-PB,fill=T["BORDER"],width=1)

        # Threshold
        if self.threshold is not None:
            ty=yt(self.threshold)
            if PT<=ty<=h-PB:
                self.create_line(PL,ty,w-PR,ty,fill=T["RED"],dash=(4,3),width=1)
                self.create_text(w-PR-2,ty-7,text=f"THR {self.threshold:.3g}",
                                 anchor="e",fill=T["RED"],font=("Consolas",7))

        def draw_series(data,color,filled=True):
            if len(data)<2: return
            pts=[(xt(i,len(data)),yt(v)) for i,v in enumerate(data)]
            if filled:
                poly=[PL,h-PB]
                for x2,y2 in pts: poly+=[x2,y2]
                poly+=[w-PR,h-PB]
                self.create_polygon(*poly,fill=self._dim(color,0.09),outline="")
            flat=[]
            for p in pts: flat+=[p[0],p[1]]
            self.create_line(*flat,fill=color,width=2,smooth=True)
            lx,ly=pts[-1]
            self.create_oval(lx-4,ly-4,lx+4,ly+4,fill=color,outline=T["BG"],width=2)

        for ov in self.overlay:
            if ov.get("data"): draw_series(list(ov["data"]),ov["color"],filled=False)
        if self.data:
            draw_series(self.data,self.color)

        # Value tag
        if self.data:
            v=self.data[-1]
            col=T["RED"] if (self.threshold and v>=self.threshold) else self.color
            self.create_text(w-PR,PT,text=f"{v:.3g}{self.unit}",
                             anchor="ne",fill=col,font=("Consolas",9,"bold"))

        # Hover
        if self._hover_x is not None and self.data and n>=2:
            idx=int((self._hover_x-PL)/max(iw,1)*(n-1))
            idx=max(0,min(idx,n-1))
            xc=xt(idx,n); val=self.data[idx]; yc=yt(val)

    def _tick(self):
        diff=self._score-self._anim; self._anim+=diff*0.18; self.redraw()
        if abs(diff)>0.2: self.after(30,self._tick)

        w,h=self.winfo_width(),self.winfo_height()

                         fill=acol,font=("Consolas",12,"bold"))

        super().__init__(parent,bg=T["PANEL2"],highlightthickness=0,height=52,**kw)
        self._events=[]; self._on_click=on_click; self._start_t=None
        self.bind("<Configure>",lambda e:self.redraw())
        self.bind("<Button-1>",self._click)

    def update_events(self,events):
        self._events=events
        if events and self._start_t is None: self._start_t=events[0]["time"]
        self.redraw()

    def _click(self,event):
        if self._on_click and self._events:
            idx=self._nearest(event.x)
            if idx is not None: self._on_click(self._events[idx])

    def _nearest(self,x):
        w=self.winfo_width()
        if not self._events or not self._start_t: return None
        span=max(time.time()-self._start_t,1); best,bd=None,999
        for i,ev in enumerate(self._events):
            ex=10+(w-20)*(ev["time"]-self._start_t)/span
            if abs(ex-x)<bd: best,bd=i,abs(ex-x)
        return best if bd<20 else None

    def redraw(self):
        self.config(bg=T["PANEL2"]); self.delete("all")
        w,h=self.winfo_width(),self.winfo_height()
        if w<20: return
        now=time.time(); start=self._start_t or now; span=max(now-start,1)
        self.create_line(10,30,w-10,30,fill=T["BORDER2"],width=1)
        for frac in [0,0.25,0.5,0.75,1.0]:

            self.create_text(x,44,text=lbl,fill=T["SUBTEXT"],font=("Consolas",6))

class PredictionEngine:
    @staticmethod
    def analyze(snap):
        if len(snap["times"])<10: return None
        result={}; t_arr=np.array(snap["times"]); t_arr-=t_arr[0]
        for key,vals in [("temp",snap["temp"]),("vibe",snap["vibe"]),
                         ("amp",snap["amp"]),("health",snap["health"])]:
            if len(vals)<5: continue
            arr=np.array(vals[-TREND_WINDOW:]); t=t_arr[-len(arr):]
            sl,ic,r,_,_=linregress(t,arr)
            result[key]={"slope":sl,"r2":r**2,"current":arr[-1]}
        preds=[]
        for ch,thr in THRESHOLDS.items():
            if ch not in result: continue
            rd=result[ch]
            if rd["slope"]>0 and rd["current"]<thr:
                rem=(thr-rd["current"])/rd["slope"]
                if 0<rem<600:
                    conf="HIGH" if rd["r2"]>0.7 else "MED" if rd["r2"]>0.4 else "LOW"
                    preds.append((ch.upper(),rem/60,rd["slope"],rd["r2"],conf))
        return {"channels":result,"predictions":preds,
                "health_slope":result.get("health",{}).get("slope",0)}

    # ── Tab switching ────────────────────────────────────────────
    def _switch_tab(self,key):
        self._active_tab.set(key)
        for k,frame in self._tab_frames.items():
            if k==key: frame.grid()
            else:      frame.grid_remove()
        for k,btn in self._tab_btns.items():
            btn.config(bg=T["CYAN"] if k==key else T["PANEL2"],
                       fg=T["BG"]   if k==key else T["SUBTEXT"])

    # ── Theme toggle ─────────────────────────────────────────────
    def _toggle_theme(self):
        self._dark_mode=not self._dark_mode
        apply_theme(DARK if self._dark_mode else LIGHT)
        self._theme_btn.config(text="☀ LIGHT" if self._dark_mode else "🌙 DARK")
        for w in self.winfo_children(): w.destroy()
        self._build_ui()

    # ── Quick actions ────────────────────────────────────────────
    def _recalibrate(self):
        self._send_cmd("RECAL")
        messagebox.showinfo("Recalibrate","Command sent.\nKeep machine still for 2 seconds.")

    def _export_csv(self):
        if not os.path.exists(LOG_FILE):
            messagebox.showwarning("Export","No log file found yet."); return
        dest=filedialog.asksaveasfilename(defaultextension=".csv",
             filetypes=[("CSV","*.csv")],
             initialfile=f"export_{datetime.datetime.now().strftime('%Y%m%d_%H%M%S')}.csv")
        if dest: shutil.copy(LOG_FILE,dest); messagebox.showinfo("Export",f"Saved to:\n{dest}")

    def _clear_log(self):
        if messagebox.askyesno("Clear Log","Delete all log data?"):
            try: self.log_f.close()
            except Exception: pass
            if os.path.exists(LOG_FILE): os.remove(LOG_FILE)
            self._setup_log(); messagebox.showinfo("Clear","Log cleared.")

    def _on_timeline_click(self,event):
        ts=datetime.datetime.fromtimestamp(event["time"]).strftime("%H:%M:%S")
        messagebox.showinfo("Event",
            f"Type:     {event['kind'].upper()}\n"
            f"Channel:  {event['key'].upper()}\n"
            f"Value:    {event['value']:.4g}\n"
            f"Severity: {event.get('severity','')}\n"
            f"Time:     {ts}")

    # ── Helpers ─────────────────────────────────────────────────
    def _set_status(self,msg,color):
        try: self.status_lbl.config(text=msg,fg=color)
        except Exception: pass

    def _refresh_ports(self):
        ports=[p.device for p in serial.tools.list_ports.comports()]
        self.port_cb["values"]=ports
        if ports: self.port_var.set(ports[0])

    def _send_cmd(self,cmd):
        if self.reader: self.reader.send_command(cmd)

    def _toggle_connect(self):
        if self.reader and self.reader.running:
            self.reader.stop(); self.reader=None
            self.store.connected=False
            self.conn_btn.config(text=" CONNECT ",bg=T["CYAN"],fg=T["BG"])
            self._set_status("◆ DISCONNECTED",T["SUBTEXT"])
        else:
            port=self.port_var.get()
            if not port: return
            self.reader=SerialReader(port,self.store,self._log_row,self._on_serial_error)
            self.reader.start()
            self.conn_btn.config(text=" DISCONNECT ",bg=T["RED"],fg=T["TEXT"])
            self._set_status("◈ CONNECTING…",T["YELLOW"])

    # ── Main Refresh ─────────────────────────────────────────────
    def _refresh(self):
        snap=self.store.get_snapshot(); now=time.time()

        if self.store.connected:
            if snap["times"]:
                age=now-snap["times"][-1]
                col=T["GREEN"] if age<4 else T["YELLOW"]
                self._set_status("◆ LIVE — RECEIVING DATA",col)
                self.sample_lbl.config(
                    text=f"samples: {len(snap['times'])}   log → {LOG_FILE}",fg=T["SUBTEXT"])
            else:
                self._set_status("◈ CONNECTED — WAITING…",T["YELLOW"])

        try: self.banner.update(snap["alerts"],now,snap)
        except Exception: pass

        if snap["health"]:
            self.gauge.set_score(snap["health"][-1])

        self.breakdown.update(snap["last"])

        last=snap["last"]
        if last:
            t,h,v,a,s=last[0],last[1],last[2],last[3],last[4]
            vals={"temp":t,"humid":h,"vibe":v,"amp":a,"sound":s}
            for key,tile in self.tiles.items():
                val=vals.get(key,0); thr=THRESHOLDS.get(key,100)
                ratio=val/thr if thr else 0
                col=T["GREEN"] if ratio<0.6 else T["YELLOW"] if ratio<0.9 else T["RED"]
                tile.val_lbl.config(text=f"{val:.3g} {tile.unit}",fg=col)

        times=snap["times"]
        humid_ov=[{"data":snap["humid"],"color":T["BLUE"],"label":"humid","unit":"%"}]

        # Overview charts
        self.charts["ov_temp"].update_data(snap["temp"],times,THRESHOLDS["temp"],humid_ov)
        self.charts["ov_vibe"].update_data(snap["vibe"],times,THRESHOLDS["vibe"])
        self.charts["ov_amp"].update_data(snap["amp"],times,THRESHOLDS["amp"])
        self.charts["ov_health"].update_data(snap["health"],times)

        # Temp tab
        self.charts["tt_temp"].update_data(snap["temp"],times,THRESHOLDS["temp"])
        self.charts["tt_humid"].update_data(snap["humid"],times)

        # Vibe tab
        self.charts["tv_vibe"].update_data(snap["vibe"],times,THRESHOLDS["vibe"])

        # Electrical tab
        self.charts["te_amp"].update_data(snap["amp"],times,THRESHOLDS["amp"])

        # Noise tab
        self.charts["tn_sound"].update_data(snap["sound"],times,THRESHOLDS["sound"])

        self.thr_panel.update_proximity(last)
        self.alert_panel.update(snap["alerts"],now)
        self.corr_panel.update(snap)
        self.timeline.update_events(snap.get("events",[]))

        pred=PredictionEngine.analyze(snap)
        if pred:
            lines=[]
            slope=pred["health_slope"]
            if slope<-0.1:   lines.append(f"⬇ Declining {abs(slope)*60:.1f} pts/min")
            elif slope>0.05: lines.append("⬆ Health improving")
            else:            lines.append("✓ Stable")
            for ch,mins,sl,r2,conf in pred["predictions"]:
                lines.append(f"▲ {ch} threshold in ~{mins:.1f} min [{conf}]")
            if not pred["predictions"] and slope>=-0.1:
                lines.append("All parameters safe.")
            self.pred_lbl.config(text="\n".join(lines),fg=T["TEXT"])

        self.after(500,self._refresh)

    def destroy(self):
        if self.reader: self.reader.stop()
        try: self.log_f.close()
        except Exception: pass
        super().destroy()
